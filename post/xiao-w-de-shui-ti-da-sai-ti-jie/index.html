T1

一个easy的二进制枚举

var
  n,m,i,s,sum:longint;//定义一堆乱七八糟的东西
  b:array[0..20] of 0..1;//定义b数组
begin
  readln(m,n);//读入m和n（反了一下）
  while b[0]=0 do//当哨兵为0的时候
    begin
      i:=n;while b[i]=1 do begin b[i]:=0;dec(i);end;b[i]:=1;//二进制枚举
      s:=0;//s清零
      for i:=1 to n do s:=s+b[i];//加一下
      if s<=m then inc(sum);//如果大王小于等于m，则方案数加一
    end;
  writeln(sum);//输出方案数
end.

一题找规律的题目

规律如下：

{f(1)=1f(2)=2f(3)=4f(4)=7f(n)=f(n−1)+f(n−2)+f(n−4)\begin{cases}f(1)=1\\f(2)=2\\f(3)=4\\f(4)=7\\f(n)=f(n-1)+f(n-2)+f(n-4)\end{cases}⎩⎪⎪⎪⎪⎪⎪⎨⎪⎪⎪⎪⎪⎪⎧​f(1)=1f(2)=2f(3)=4f(4)=7f(n)=f(n−1)+f(n−2)+f(n−4)​

知道了递推公式，下面就是高精加法了

var
  a,b,c,d,e:array[0..3000] of byte;//定义abcde数组
  n,i,j,k:longint;//定义别的东西
begin
  readln(n);//读入n
  if n=1 then begin writeln(1);exit;end;
  if n=2 then begin writeln(2);exit;end;
  if n=3 then begin writeln(4);exit;end;
  if n=4 then begin writeln(7);exit;end;
  //以上为特判
  a[1]:=1;b[1]:=2;c[1]:=4;d[1]:=7;//初始值
  for i:=5 to n do//循环
    begin
      k:=0;//进位清零
      for j:=1 to 3000 do//每个位数做一次
        begin
          e[j]:=a[j]+c[j]+d[j]+k;//加起来
          k:=e[j] div 10;e[j]:=e[j] mod 10;//处理进位
        end;
      a:=b;b:=c;c:=d;d:=e;//交换
    end;
  k:=3000;while (e[k]=0) and (k>1) do dec(k);//查找非零的位数
  for i:=k downto 1 do write(e[i]);//输出
end.

T3

一个简单而且代码很短的模板题。

var
  a,b,c:array[0..2001] of longint;
  s,s1,s2:ansistring;
  x:char;
  l,i:longint;
 procedure jia;//高精加
  var
    i:longint;
  begin
    for i:=1 to 400 do
      begin
        c[i]:=a[i]+b[i]+l;
        l:=c[i] div 10;
        c[i]:=c[i] mod 10;
      end;
  end;
 procedure jian;//高精减
  var
    i:longint;
  begin
    for i:=1 to 400 do
      if a[i]-b[i]-l>=0 then
        begin
          c[i]:=a[i]-b[i]-l;
          l:=0;
        end
        else
          begin
            c[i]:=a[i]-b[i]-l+10;
            l:=1;
          end;
  end;
 procedure cheng;//高精乘
  var
    i,j:longint;
  begin
    for i:=1 to 1000 do
      for j:=1 to 1000 do
        begin
          c[i+j-1]:=c[i+j-1]+a[i]*b[j]+l;
          l:=c[i+j-1] div 10;
          c[i+j-1]:=c[i+j-1] mod 10;
        end;
  end;
begin
  readln(s);//读入s
  if pos('-',s)>0 then//如果是减法
    begin
      x:='-';s:=s+'-';
      s1:=copy(s,1,pos('-',s)-1);delete(s,1,pos('-',s));
      s2:=copy(s,1,pos('-',s)-1);delete(s,1,pos('-',s));
    end;
  if pos('+',s)>0 then//如果是加法
    begin
      x:='+';s:=s+'+';
      s1:=copy(s,1,pos('+',s)-1);delete(s,1,pos('+',s));
      s2:=copy(s,1,pos('+',s)-1);delete(s,1,pos('+',s));
    end;
  if pos('*',s)>0 then//如果是乘法
    begin
      x:='*';s:=s+'*';
      s1:=copy(s,1,pos('*',s)-1);delete(s,1,pos('*',s));
      s2:=copy(s,1,pos('*',s)-1);delete(s,1,pos('*',s));
    end;
  l:=length(s1)for i:=1 to l do val(s1[l-i+1],a[i]);//转换到数组里
  l:=length(s2);for i:=1 to l do val(s2[l-i+1],b[i]);//同理
  l:=0;
  case x of//加减乘运算
    '-':jian;
    '+':jia;
    '*':cheng;
  end;
  l:=2000;
  while (c[l]=0) and (l>1) do dec(l);//找到不是0的一位数
  for i:=l downto 1 do write(c[i]);//输出
end.
